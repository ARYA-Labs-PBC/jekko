name: jankurai

on:
  pull_request:
  push:
    branches: [main]

jobs:
  audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0
      - name: Install jankurai
        run: cargo install --git https://github.com/neverhuman/jankurai --tag v1.3.0 --locked jankurai
      - name: Run audit
        run: bash ops/ci/jankurai.sh || true
      - name: Enforce audit gate
        run: node tools/jankurai-audit-gate.mjs target/jankurai/repo-score.json
      - name: Stage results
        if: always()
        run: bash ops/ci/jankurai-stage.sh || true
      - name: Upload artifacts
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: jankurai-audit
          path: |
            target/jankurai/repo-score.json
            target/jankurai/repo-score.md
            target/jankurai/repair-queue.jsonl
            target/jankurai/jankurai.sarif
          if-no-files-found: ignore
